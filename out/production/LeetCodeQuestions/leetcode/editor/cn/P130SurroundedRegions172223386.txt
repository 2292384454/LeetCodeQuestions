    class Solution {
        char[][] copy;
        int m, n;

        public void solve(char[][] board) {
            if (board == null || board.length == 0) return;
            m = board.length;
            n = board[0].length;
            copy = new char[m][n];
            for (int i = 0; i < m; i++) {
                System.arraycopy(board[i], 0, copy[i], 0, m);
                Arrays.fill(board[i], 'X');
            }
            for (int i = 0; i < m; i++) {
                if (copy[i][0] == 'O') BFS(board, i, 0);
                if (copy[i][n - 1] == 'O') BFS(board, i, n - 1);
            }
            for (int j = 0; j < n; j++) {
                if (copy[0][j] == 'O') BFS(board, 0, j);
                if (copy[m - 1][j] == 'O') BFS(board, m - 1, j);
            }
        }

        public void BFS(char[][] board, int i, int j) {
            board[i][j] = 'O';
            if (i - 1 >= 0) {
                if (j - 1 >= 0 && board[i - 1][j - 1] == 'O') BFS(board, i - 1, j - 1);
                if (j + 1 < n && board[i - 1][j + 1] == 'O') BFS(board, i - 1, j + 1);
            }
            if (i + 1 < m) {
                if (j - 1 >= 0 && board[i + 1][j - 1] == 'O') BFS(board, i + 1, j - 1);
                if (j + 1 < n && board[i + 1][j + 1] == 'O') BFS(board, i + 1, j + 1);
            }
        }
